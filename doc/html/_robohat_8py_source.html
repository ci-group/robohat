196 def get_servo_is_connected(self, _servo_nr: int) -> bool:
197 """!
198 Checks if servo is connected. Returns False when not connected
199
200 @param _servo_nr The servo nr
201 @return: Returns False when not connected
202 """
203
204 if _servo_nr >= 1 or _servo_nr <= 16:
205 if self.__servo_assembly_1 is None:
206 return False
207
208 servo_nr = self.__get_servo_nr_depending_assembly(_servo_nr)
209 return self.__servo_assembly_1.get_servo_is_connected(servo_nr)
210
211 elif _servo_nr >= 17 or _servo_nr <= 32:
212 if self.__servo_assembly_2 is None:
213 return False
214
215 servo_nr = self.__get_servo_nr_depending_assembly(_servo_nr)
216 return self.__servo_assembly_2.get_servo_is_connected(servo_nr)
217 else:
218 return False

316 def __get_servo_nr_depending_assembly(self, _servo_nr: int) -> int:
317 """!
318 Get servo nr depending on the assembly
319
320 @param _servo_nr The servo nr wanted (starts at 1)
321 @return the servo nr of the assembly (so servo nr 17 will be servo nr 1 of assembly 2
322 @raises Exception, when servo nr is not valid
323 """
324
325 if _servo_nr >= 1 and _servo_nr <= 16:
326 return _servo_nr
327 elif _servo_nr >= 17 and _servo_nr <= 32:
328 return _servo_nr - 16
329 else:
330 raise Exception("Requested servo nr is not available")
