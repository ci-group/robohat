427 def __get_servo_nr_depending_assembly(self, _servo_nr: int) -> int | None:
428 """!
429 Get the servo nr of the assembly (so servo nr 16 will be servo nr 0 of assembly 2. If not available
430 None will be returned
431
432 @param _servo_nr The servo nr wanted (starts at 1)
433 @return servo number or None
434 """
435
436 if _servo_nr >= 0 and _servo_nr < 16:
